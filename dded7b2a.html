二、gitlab-server主机安装

  • 配置gitlab网络镜像源
[root@gitlab-server ~]# vim /etc/yum.repos.d/gitlab.repo      #编写网络镜像源
[gitlab-ce]
name=gitlab
baseurl=https://mirrors.tuna.tsinghua.edu.cn/gitlab-ce/yum/el7/
enabled=1
gpgcheck=0
[root@gitlab-server ~]# yum clean all                         #删除安装软件缓存
[root@gitlab-server ~]# yum repolist                          #刷新网络安装源
  • 部署gitlab服务器
[root@gitlab-server ~]# yum install -y gitlab-ce              #安装gitlab-ce
[root@gitlab-server ~]# vim +32 /etc/gitlab/gitlab.rb         #修改配置文件
external_url 'http://192.168.127.100'                         #修改成自己的客户机的IP地址
[root@gitlab-server ~]# gitlab-ctl  reconfigure               #初始化gitlab环境,需要的时间比较长，请耐心等待
[root@gitlab-server ~]# gitlab-ctl  status                    #查看gitlab服务组件状态，都是run说明服务组件运行正常
